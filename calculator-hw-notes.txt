var defenseDeck = [
  1,2,3,4,5,6,7
];
// // randomly picks an INDEX from the defense deck
// Math.floor(Math.random() * defenseDeck.length)
var attackDeck = [
  8, 14, 23, 11, 9
];
// // randomly picks an INDEX from the defense deck
// Math.floor(Math.random() * attackDeck.length)

document.querySelector("button").addEventListener("click", function(event){
  event.preventDefault();
  var choiceDeck = [];
  if(document.querySelector("#defense").checked){
    // choiceDeck.push(...defenseDeck)
    for(var i = 0; i < defenseDeck.length; i++){
      choiceDeck.push(defenseDeck[i]);
    }
  }
  if(document.querySelector("#attack").checked){
    // choiceDeck.push(...attackDeck)
    for(var i = 0; i < attackDeck.length; i++){
      choiceDeck.push(attackDeck[i]);
    }
  }
  console.log(choiceDeck);

  // // randomly picks an INDEX from the choice deck
  var choice = Math.floor(Math.random() * choiceDeck.length);
  console.log(choiceDeck[choice]);

})
